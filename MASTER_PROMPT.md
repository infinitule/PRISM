```python
def load_csv_dataset(path: str, feature_cols: list[str], label_col: str,
                     class_map: dict[str, int]) -> tuple[np.ndarray, np.ndarray]:
    """Pure-numpy CSV loader. Handles missing values, auto-normalizes."""
    rows = []
    with open(path, 'r') as f:
        header = f.readline().strip().split(',')
        feat_idx = [header.index(c) for c in feature_cols]
        lbl_idx  = header.index(label_col)
        for line in f:
            vals = line.strip().split(',')
            if any(v in ('', 'NA', 'null', 'NaN') for v in vals):
                continue  # skip rows with missing values
            x = np.array([float(vals[i]) for i in feat_idx])
            rows.append((x, class_map[vals[lbl_idx]]))
    X = np.array([r[0] for r in rows])
    Y = np.eye(len(class_map))[[r[1] for r in rows]]
    return X, Y
```

```python
def detect_domain_shift(X_synthetic: np.ndarray, X_real: np.ndarray,
                         threshold=2.0) -> dict:
    """Compute feature-wise z-score of real data mean against synthetic distribution."""
    syn_mean = X_synthetic.mean(axis=0)
    syn_std  = X_synthetic.std(axis=0) + 1e-8
    z_scores = np.abs((X_real.mean(axis=0) - syn_mean) / syn_std)
```

### T3 ◯ — ONLINE CONTINUAL LEARNING (EWC)
**Definition**: Add streaming update mode — the fiber NN updates weights incrementally as new labeled samples arrive without full retraining, using Elastic Weight Consolidation (EWC) to prevent catastrophic forgetting of previously learned tasks.

**Why it matters for the frontier**: Static models die in production. Fiber links age, amplifiers drift, new equipment is installed. A photonic NN that adapts without forgetting is the difference between a proof-of-concept and a real network management tool. EWC has a direct physical interpretation: Fisher information = which weights are "load-bearing" in the optical ring — changing them destroys the interference pattern.

**Implementation sketch**:
```python
class OnlineFiberTrainer(CrossEntropyTrainer):
    """
    EWC-regularized online trainer.
    After each task, compute Fisher information for each weight.
    New tasks are trained with L2 penalty toward old weights, scaled by Fisher info.
    Physical interpretation: high-Fisher weight → load-bearing optical phase;
    EWC prevents phase drift that would destroy interference patterns.
    """
    def __init__(self, layer_sizes, lr=5e-4, l2=1e-4, ewc_lambda=1000.0):
        super().__init__(layer_sizes, lr=lr, l2=l2)
        self.ewc_lambda = ewc_lambda
        self.fisher: list[np.ndarray] = []   # Fisher info per layer
        self.W_star: list[np.ndarray] = []   # consolidated weights

    def compute_fisher(self, X, Y, n_samples=200):
        """Estimate diagonal Fisher information matrix via sampling."""
        rng  = np.random.default_rng(0)
        idx  = rng.choice(len(X), min(n_samples, len(X)), replace=False)
        F    = [np.zeros_like(w) for w in self.W]
        for i in idx:
            y_hat = self.forward(minmax_norm(X[i]))
            log_p = np.log(y_hat + 1e-12)
            sample_class = rng.choice(len(Y[i]), p=y_hat)
            y_one_hot = np.zeros_like(Y[i]); y_one_hot[sample_class] = 1.0
            gW, _ = self.backward_ce(y_one_hot)
            for l in range(len(self.W)):
                F[l] += gW[l] ** 2
        self.fisher  = [f / len(idx) for f in F]
        self.W_star  = [w.copy() for w in self.W]

    def ewc_penalty(self) -> float:
        """EWC regularization loss contribution."""
        if not self.fisher:
            return 0.0
        return 0.5 * self.ewc_lambda * sum(
            float(np.sum(f * (w - ws)**2))
            for f, w, ws in zip(self.fisher, self.W, self.W_star)
        )

    def backward_ewc(self, y_true) -> tuple:
        """CE gradient + EWC gradient."""
        gW, gb = self.backward_ce(y_true)
        if self.fisher:
            for l in range(len(gW)):
                gW[l] += self.ewc_lambda * self.fisher[l] * (self.W[l] - self.W_star[l])
        return gW, gb

    def online_update(self, x_raw, y_true, n_steps=5):
        """Single-sample online update with EWC regularization."""
        x_n = minmax_norm(x_raw)
        for _ in range(n_steps):
            self.forward(x_n)
            gW, gb = self.backward_ewc(y_true)
            self._adam(gW, gb)
```

### T5 ◯ — MULTI-NODE WDM OPTICAL MESH
**Definition**: Simulate a WDM ring topology with N PRISM nodes, each a coherent fiber NN, passing activations over optical links with realistic propagation delay (c/n_eff) and loss (α·L). This is a photonic graph neural network — nodes are optical processors, edges are fiber spans.

**Why it matters**: All current PRISM models are single-node. Real optical networks are graphs. A multi-node mesh enables: (1) distributed inference where each node classifies its local segment; (2) consensus inference where node outputs are aggregated over optical links; (3) simulation of real metro/regional ring topologies (ROADM-based 4-8 node rings).

**Implementation sketch**:
```python
# prism_mesh.py — new file
import numpy as np
from dataclasses import dataclass, field
from typing import Optional
from fiber_physics import SMF_ALPHA_DB_KM, C_LIGHT

@dataclass
class FiberSpan:
    """One WDM optical link between two PRISM nodes."""
    length_km: float
    alpha_db_km: float = SMF_ALPHA_DB_KM
    n_eff: float = 1.4676   # effective refractive index at 1550 nm

    @property
    def propagation_delay_ns(self) -> float:
        return self.length_km * 1e3 / (C_LIGHT / self.n_eff) * 1e9

    @property
    def field_loss(self) -> float:
        loss_db = self.alpha_db_km * self.length_km
        return 10 ** (-loss_db / 20)  # amplitude loss factor

    def propagate(self, activation: np.ndarray) -> np.ndarray:
        """Attenuate activation vector over fiber span (per-channel)."""
        return activation * self.field_loss

@dataclass
class PRISMNode:
    """One node in the optical mesh — a CoherentFiberNetwork with ID and position."""
    node_id: str
    fiber_nn: object              # CoherentFiberNetwork
    class_names: list[str]
    neighbors: dict = field(default_factory=dict)  # {neighbor_id: FiberSpan}

    def infer(self, x: np.ndarray) -> np.ndarray:
        return self.fiber_nn.forward(x)

    def add_link(self, neighbor_id: str, span: FiberSpan):
        self.neighbors[neighbor_id] = span

class WDMRingMesh:
    """
    WDM ring of N PRISM nodes connected by fiber spans.
    Activations propagate around the ring with realistic delay and loss.
    Consensus: each node's output is attenuated and fed to the next node.
    """
    def __init__(self, nodes: list[PRISMNode], ring_spans: list[FiberSpan]):
        assert len(nodes) == len(ring_spans), "One span per inter-node link"
        self.nodes = {n.node_id: n for n in nodes}
        self.ordered = [n.node_id for n in nodes]
        # Wire ring: node[i] → span[i] → node[(i+1)%N]
        for i, node in enumerate(nodes):
            next_id = nodes[(i+1) % len(nodes)].node_id
            node.add_link(next_id, ring_spans[i])
        self.total_delay_ns = sum(s.propagation_delay_ns for s in ring_spans)
        self.total_loss_db  = sum(s.alpha_db_km * s.length_km for s in ring_spans)

    def consensus_infer(self, x: np.ndarray, rounds: int = 2) -> dict:
        """
        Distribute input around the ring for consensus inference.
        Each round: every node infers, attenuates output, passes to next.
        Final: average all node outputs (weighted by inverse span loss).
        """
        outputs = {}
        activation = x.copy()
        for _ in range(rounds):
            for nid in self.ordered:
                node  = self.nodes[nid]
                out   = node.infer(activation)
                outputs[nid] = out
                # Pass attenuated output to next node's input space
                # (dimension may differ — project via mean pooling)
                next_id = node.neighbors and list(node.neighbors.keys())[0]
                if next_id:

                    # Pad/truncate to match next node's input dimension
                    n_in = self.nodes[next_id].fiber_nn.layers[0].n_in
                    if len(propagated) < n_in:
                        activation = np.pad(propagated, (0, n_in - len(propagated)))
                    else:
                        activation = propagated[:n_in]
        # Consensus: average over all nodes (equal weight for now)
        all_preds = np.array(list(outputs.values()))
        return {
            'consensus':    all_preds.mean(axis=0),
            'node_outputs': outputs,
            'delay_ns':     self.total_delay_ns,
            'loss_db':      self.total_loss_db,
```

```python
@dataclass
class AgentStep:

    thought: str
    action: str
    observation: str

class PRISMAgent:
    """
    ReAct-style agentic loop for recursive architecture self-improvement.
    THINK → ACT → OBSERVE → THINK → ...
    Terminates when target_acc reached or max_steps exhausted.
    """
    ACTION_SPACE = ['DEEPEN', 'WIDEN', 'RETRAIN', 'LOWER_LR', 'AUGMENT_DATA',
                    'RECALIBRATE', 'HALT']

    def __init__(self, engine, target_acc=0.99, max_steps=20):
        self.engine     = engine
        self.target_acc = target_acc
        self.max_steps  = max_steps
        self.trace: list[AgentStep] = []
        self.best_acc   = 0.0
        self.stall_count = 0

    def _think(self, acc: float, arch: list[int], prev_acc: float) -> tuple[str, str]:
        """Rule-based policy → next action and rationale."""
        delta = acc - prev_acc
        if acc >= self.target_acc:
            return 'HALT', f"Target {self.target_acc:.0%} reached at {acc:.1%}"
        if delta < 0.005 and self.stall_count >= 2:
            return 'LOWER_LR', f"Stalled ({delta:+.3f} over 3 steps) — halve lr"
        if acc < 0.70:
            return 'AUGMENT_DATA', f"Low accuracy {acc:.1%} — expand dataset"
        if len(arch) <= 3:
            return 'DEEPEN', f"Shallow arch {arch} — insert fiber ring loop"
        if delta < 0.01:
            return 'WIDEN', f"Marginal gain {delta:+.3f} — expand narrowest layer"
        return 'RETRAIN', f"Moderate gain {delta:+.3f} — continue training"

    def _act(self, action: str) -> float:
        """Execute action on the engine, return new fiber accuracy."""
        if action == 'DEEPEN':
            self.engine.expand_depth()
        elif action == 'WIDEN':
            self.engine.expand_width()
        elif action == 'RETRAIN':
            self.engine.retrain(epochs=200)
        elif action == 'LOWER_LR':
            self.engine.lr *= 0.5
            self.engine.retrain(epochs=200)
        elif action == 'AUGMENT_DATA':
            self.engine.augment_dataset(factor=2)
            self.engine.retrain(epochs=300)
        elif action == 'RECALIBRATE':
            self.engine.recalibrate()
        return self.engine.current_fiber_acc()

    def run(self):
        acc = self.engine.current_fiber_acc()
        prev_acc = 0.0
        for step in range(self.max_steps):
            arch = list(self.engine.current_arch)
            thought, rationale = self._think(acc, arch, prev_acc)
            if thought == 'HALT':
                self.trace.append(AgentStep(step, rationale, 'HALT',
                                            f"Final acc={acc:.1%}", acc, arch))
                break
            new_acc = self._act(thought)
            obs = (f"acc: {acc:.1%} → {new_acc:.1%} ({new_acc-acc:+.3f})  "
                   f"arch: {arch}")
            if abs(new_acc - acc) < 0.001:
                self.stall_count += 1
            else:
                self.stall_count = 0
            self.trace.append(AgentStep(step, rationale, thought, obs, new_acc, arch))
            prev_acc, acc = acc, new_acc
        return self.trace

    def print_trace(self):
        print(f"\n{'═'*62}")
        print(f"  PRISM AGENT — DECISION TRACE  ({len(self.trace)} steps)")
        print(f"{'═'*62}")
        for s in self.trace:

            print(f"  THOUGHT : {s.thought}")
            print(f"  OBS     : {s.observation}")
            print(f"  Acc     : {s.fiber_acc*100:.1f}%  Arch: {s.arch}")
```

### T8 ◯ — QUANTUM EXTENSION: SQUEEZED LIGHT
**Definition**: Add squeezed-light mode to the fiber physics layer. Model shot-noise reduction via squeezing parameter r, compute quantum Fisher information (QFI) for each layer, compare classical vs quantum SNR. Show the regime where quantum advantage exists.

**Why it matters**: This elevates PRISM from classical photonics to quantum photonics — one of the most active research frontiers. Squeezed light is already used in gravitational wave detection (LIGO) and quantum key distribution. Applying it to optical neural networks is novel and publishable.

**Physical model**:
Squeezed vacuum state: variance in one quadrature reduced below shot noise limit.
  V_x = e^{-2r}/4,   V_p = e^{+2r}/4   (two-mode squeezing, r = squeezing parameter)
Standard quantum limit: V_SQL = 1/4 (for coherent state, r=0)
SNR improvement: ΔSNR = 10 log₁₀(e^{2r}) dB

Quantum Fisher information (QFI) for phase estimation:
  F_Q = 4 · Var(Ĥ)   where Ĥ is the phase-shift generator
  For N photons + squeezing:  F_Q = N²·sinh²(r) + N·cosh(2r)

**Implementation sketch**:
```python
# In fiber_physics.py — add QuantumReceiver
class QuantumHomodyneReceiver:
    """
    Homodyne receiver with squeezed local oscillator.
    Reduces shot noise below standard quantum limit.
    """
    def __init__(self, responsivity=0.9, squeezing_r=0.0):
        self.R = responsivity
        self.r = squeezing_r   # squeezing parameter (0 = coherent, >0 = squeezed)

    @property
    def shot_noise_variance(self):
        """Variance in squeezed quadrature: V = e^{-2r}/4"""
        return np.exp(-2 * self.r) / 4

    @property
    def snr_improvement_db(self):
        """SNR improvement over coherent detection: 20r/ln(10) dB"""
        return 20 * self.r / np.log(10)

    def detect(self, E_signal, E_LO=1.0, n_photons=1e6, rng=None):
        I = self.R * E_LO * float(np.real(E_signal))
        if rng is not None:
            # Squeezed noise: reduced by e^{-2r} vs shot noise
            sigma = np.sqrt(self.shot_noise_variance / n_photons)
            I += rng.normal(0, sigma)
        return I

    def quantum_fisher_information(self, n_photons):
        """QFI for phase estimation with squeezed input."""
        N = n_photons
        r = self.r
        return N**2 * np.sinh(r)**2 + N * np.cosh(2*r)

    def classical_fisher_information(self, n_photons):
        """Classical FI (coherent state, r=0): F_C = N"""
        return n_photons

    def quantum_advantage(self, n_photons):
        """Ratio QFI/CFI — >1 means quantum advantage."""
        return (self.quantum_fisher_information(n_photons) /
                self.classical_fisher_information(n_photons))
```

### T11 ◯ — NEUROMORPHIC SPIKING OPTICAL NEURONS
**Definition**: Replace continuous IQ neurons with spiking optical neurons. Each neuron fires a pulse when accumulated photocurrent exceeds a threshold. Information encoded in spike timing (time-to-first-spike). Enables asynchronous optical inference — no clock, latency scales with signal strength.

**Why it matters**: Spiking neural networks are the third wave of neural network research. Spiking photonic neurons are a real experimental platform (IBM, NTT). PRISM adding spike mode creates a simulation bridge between rate-coded and spike-coded optical neural computation.

**Implementation sketch**:
```python
class SpikingOpticalNeuron:
    """
    Leaky integrate-and-fire optical neuron.
    Membrane potential integrates photocurrent until threshold → spike.
    Reset after spike. Refactory period = 1 time step.
    Physical: threshold corresponds to saturation power of the detector.
    """
    def __init__(self, threshold=1.0, tau_m=10.0, dt=0.1):
        self.threshold  = threshold
        self.tau_m      = tau_m   # membrane time constant (in dt units)
        self.dt         = dt
        self.reset()

    def reset(self):
        self.V_m        = 0.0    # membrane potential
        self.t_last_spike = -999
        self.spike_times  = []

    def step(self, I_input, t):
        """
        Euler integration of LIF dynamics:
          dV/dt = -V/τ_m + I(t)
        Returns True if neuron spikes at time t.
        """
        if t - self.t_last_spike < 1.0:  # refractory period
            return False
        self.V_m += self.dt * (-self.V_m / self.tau_m + I_input)
        if self.V_m >= self.threshold:
            self.spike_times.append(t)
            self.t_last_spike = t
            self.V_m = 0.0  # hard reset
            return True
        return False

    def time_to_first_spike(self, I_input, max_t=100.0):
        """Encode input current as latency to first spike."""
        self.reset()
        for t_idx in range(int(max_t / self.dt)):
            t = t_idx * self.dt
            if self.step(I_input, t):
                return t
        return max_t  # no spike → encode as max latency
```

```python
import socket, json
from dataclasses import dataclass

@dataclass
class InstrumentConfig:
    host: str; port: int; timeout: float = 5.0

class SCPIBridge:
    """
    SCPI socket bridge for optical instruments (OSA, BERT, PM).
    Compatible with: Keysight N7744A, JDSU MTS-8000, Viavi ONX-220.
    """
    def __init__(self, config: InstrumentConfig):
        self.cfg = config
        self._sock = None

    def connect(self):
        self._sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self._sock.settimeout(self.cfg.timeout)
        self._sock.connect((self.cfg.host, self.cfg.port))

    def send(self, cmd: str) -> str:
        self._sock.sendall((cmd + '\n').encode())
        return self._sock.recv(4096).decode().strip()

    def measure_osnr(self) -> float:
        return float(self.send('MEAS:OPT:OSNR?'))

    def measure_power_dbm(self) -> float:
        return float(self.send('MEAS:POW?'))

    def disconnect(self):
        if self._sock:
            self._sock.close()

class HITLSession:
    """
    Hardware-in-the-loop session: PRISM infers → instrument measures → adapt.
    If no instrument available, runs in simulation mode with Gaussian noise.
    """
    def __init__(self, fiber_nn, class_names, instrument: SCPIBridge = None):
        self.nn = fiber_nn; self.classes = class_names
        self.instrument = instrument
        self.adaptation_log = []

    def run_cycle(self, x_synthetic: list[float]) -> dict:
        x = np.array(x_synthetic)
        pred_probs = self.nn.forward(x)
        pred_class = self.classes[int(np.argmax(pred_probs))]

        # Get real measurement (or simulate)
        if self.instrument:
            real_osnr  = self.instrument.measure_osnr()
            real_power = self.instrument.measure_power_dbm()
            real_data  = np.array([real_osnr, real_power] + x_synthetic[2:].tolist())
        else:
            real_data = x + np.random.normal(0, 0.1, x.shape)  # sim mode

        result = {
            'input':           x_synthetic,
            'predicted_class': pred_class,
            'confidence':      float(np.max(pred_probs)),
            'real_data':       real_data.tolist(),
            'mode':            'hardware' if self.instrument else 'simulation',
        }
        self.adaptation_log.append(result)
        return result
```
